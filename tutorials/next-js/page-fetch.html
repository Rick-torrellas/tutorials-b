Fetch data en nextjs, pages

Como hacer llamadas y extraer data, este procedimiento nada mas puede usarse para los componentes, declarados en la carpeta pages.


        import axios from "axios";
        function Content({ res }) {

            return (
              <>
                <div width="5%" />
                {res.map((Res, i) => {
                  return <p key={i} >{Res.name}</p>
                })}
              </>
            );
          }
        export async function getServerSideProps() {
            const res = await axios.get(`${config.host}/api/data/home`);
            return {
              props: { res: res.data },
            };
          }
