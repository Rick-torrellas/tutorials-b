        $args['meta_query'][] =  array(

            'compare' => '
            <='
        );

    }
    if(isset($attributes['price_above']))
    {
        $args['meta_query'][] =  array(
            'key' => 'price',
            'value' => $attributes['price_above'],
            'type' => 'numeric',
            'compare' => '>='
        );

    }


    if( isset($attributes['colour']) )
    {

        $args['meta_query'][] =  array(
            'key' => 'colour',
            'value' => $attributes['colour'],
            'compare' => '='
        );

            'taxonomy' => 'brands',

        ];
