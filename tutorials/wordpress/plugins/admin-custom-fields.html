Crear custom fields

Register Settings

register_settings($setting["option_group"],$settings["option_name",$setting["callback"]]);

Add Settings Section

add_settings_section($section["id"],$section["title"],(isset($section["callback"]) ? $section["callback"] : ''),$section["page"]);

Add Settings Field

add_settings_field($field["id"],$field["title"],(isset($field["callback"]) ? $field["callback"] : ''),$field["section"],(isset($field["arg"]) ? $field["arg"] : ''));

Anclarlo al hook

add_action('admin_init',[$this,'registerCustomFields']);
